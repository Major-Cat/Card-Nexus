class Card {
	private value
	private suit
	private face
	Private procedure make_face {
		if value == 1 { 
			self.face = “A”
		}
		else if value == 11 { 
			self.face == “J”
		}
		else if value == 12 {
			self.face == “Q”
		}
		else if value == 13{
			self.face ==  “K”
		}
	}
}

class Deck {
	private Cards
	private suits
	private no_cards
	for i in suits{
		for j in no_cards{
		Cards.append(Card(i, j))
		}
	}
	
	Private procedure shuffle{
		private new_deck
		for i in Cards{
			next_card = random.choice(Cards)
			new_deck.append(next_card)
			Cards.pop(next_card)
		}
		Cards = new_deck
	}
	
	Private function draw{
		Drawn = Cards[0]
		Cards.remove(Drawn)
		return Drawn
	}
}
